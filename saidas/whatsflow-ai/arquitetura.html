04 — Banco de Dados
Schema Supabase / PostgreSQL
Multi-tenant com Row Level Security (RLS). Toda query é automaticamente filtrada por tenant_id.
Diagrama de Entidades
tenants ──────────────────────────────────────────────────┐
  │                                                         │
  ├─── users (role: owner|admin|member)                    │
  ├─── whatsapp_instances                                  │
  ├─── ai_configuration                                    │
  ├─── knowledge_base ──► knowledge_chunks (pgvector)      │
  ├─── follow_up_sequences ──► follow_up_steps             │
  └─── contacts ─────────────────────────────────────────┐ │
         │                                               │ │
         ├─── conversations ──► messages                 │ │
         ├─── qualification_answers                      │ │
         ├─── follow_up_enrollments                      │ │
         └─── scheduled_meetings                         │ │
                                                         │ │
subscriptions (Stripe) ──────────────────────────────────┘ │
audit_logs ─────────────────────────────────────────────────┘
tenants
create table tenants (
  id          uuid primary key default gen_random_uuid(),
  name        text not null,
  slug        text unique not null,
  plan        text default 'starter',
  -- starter|pro|business|agency
  ai_msgs_used  int default 0,
  ai_msgs_limit int default 500,
  wa_limit      int default 1,
  stripe_customer_id text,
  stripe_sub_id      text,
  settings    jsonb default '{}',
  created_at  timestamptz default now()
);
contacts
create table contacts (
  id          uuid primary key default gen_random_uuid(),
  tenant_id   uuid references tenants,
  phone       text not null,
  name        text,
  profile_pic text,
  lead_score  int default 0,  -- 0-100
  lead_status text default 'cold',
  -- cold | warm | hot
  crm_stage   text default 'new',
  -- new|attending|qualified|proposal
  -- |negotiation|closed|lost
  tags        text[],
  custom_fields jsonb default '{}',
  last_seen   timestamptz,
  created_at  timestamptz default now(),
  unique(tenant_id, phone)
);
conversations + messages
create table conversations (
  id              uuid primary key default gen_random_uuid(),
  tenant_id       uuid references tenants,
  contact_id      uuid references contacts,
  instance_id     uuid references whatsapp_instances,
  status          text default 'active',
  -- active|closed|bot|human
  ai_enabled      bool default true,
  unread_count    int default 0,
  last_message    text,
  last_message_at timestamptz
);

create table messages (
  id              uuid primary key default gen_random_uuid(),
  tenant_id       uuid references tenants,
  conversation_id uuid references conversations,
  direction       text,  -- inbound|outbound
  content         text,
  type            text default 'text',
  sent_by         text,  -- ai|human|system
  status          text default 'sent',
  evo_message_id  text,
  created_at      timestamptz default now()
);
knowledge_base + chunks (RAG)
create table knowledge_base (
  id          uuid primary key default gen_random_uuid(),
  tenant_id   uuid references tenants,
  title       text,
  file_name   text,
  file_type   text,  -- pdf|docx|txt
  file_url    text,
  status      text default 'processing',
  created_at  timestamptz default now()
);

-- Requer: create extension vector;
create table knowledge_chunks (
  id                uuid primary key default gen_random_uuid(),
  knowledge_base_id uuid references knowledge_base,
  tenant_id         uuid references tenants,
  content           text,
  embedding         vector(1536),  -- OpenAI ada-002
  metadata          jsonb
);

-- Index para busca vetorial
create index on knowledge_chunks
  using ivfflat (embedding vector_cosine_ops);
follow_up_sequences + steps
create table follow_up_sequences (
  id          uuid primary key default gen_random_uuid(),
  tenant_id   uuid references tenants,
  name        text,
  trigger     text,
  -- no_response|proposal_sent|lead_created
  is_active   bool default true
);

create table follow_up_steps (
  id           uuid primary key default gen_random_uuid(),
  sequence_id  uuid references follow_up_sequences,
  step_number  int,
  delay_days   int,
  message_tpl  text  -- {nome}, {empresa}, etc.
);

create table follow_up_enrollments (
  id           uuid primary key default gen_random_uuid(),
  tenant_id    uuid references tenants,
  contact_id   uuid references contacts,
  sequence_id  uuid references follow_up_sequences,
  current_step int default 0,
  status       text default 'active',
  next_send_at timestamptz
);
RLS Policies (Multi-tenant)
-- Ativar RLS em todas as tabelas
alter table contacts enable row level security;
alter table conversations enable row level security;
alter table messages enable row level security;
-- (repetir para todas)

-- Policy padrão: usuário só vê dados do seu tenant
create policy tenant_isolation on contacts
  using (
    tenant_id = (
      select tenant_id
      from users
      where id = auth.uid()
    )
  );

-- Função helper para pegar tenant do usuário
create function get_tenant_id()
returns uuid as $$
  select tenant_id
  from users
  where id = auth.uid()
$$ language sql stable security definer;

Prompt Builder — Como a IA é contextualizada
// packages/shared/utils/prompt-builder.ts
export function buildSystemPrompt(config: AIConfiguration, chunks: KnowledgeChunk[]) {
  return `
Você é um assistente de vendas e atendimento da empresa ${config.company_name}.

SOBRE A EMPRESA:
${config.about}

SERVIÇOS E PRODUTOS:
${config.services}

HORÁRIO DE FUNCIONAMENTO:
${JSON.stringify(config.business_hours)}

POLÍTICA COMERCIAL:
${config.commercial_policy}

CONHECIMENTO RELEVANTE (base vetorial):
${chunks.map(c => c.content).join('\n\n')}

INSTRUÇÕES DE COMPORTAMENTO:
- Seja ${config.ai_tone} no tom de voz
- Nunca invente informações — use apenas o que está no contexto
- Quando não souber, diga que vai verificar e ofereça continuar
- Qualifique o lead naturalmente durante a conversa
- Sempre busque próximo passo (reunião, proposta, demonstração)
- Responda em até 3 parágrafos curtos
- Use linguagem natural, nunca robotizada

PERSONA: ${config.ai_persona}
  `;
}

07 — Roadmap MVP
10 Semanas para o Lançamento
Cada fase entrega valor incrementalmente. Ao final de cada fase, o produto é demonstrável para clientes beta.
1
Fundação — Auth, DB, Layout
Semanas 1–2
Supabase: schema completo + RLS policies + tipos TypeScript gerados
Auth: login, registro, convite de equipe, middleware de proteção
Layout do dashboard: sidebar, topbar, navegação responsiva
Página de overview com métricas (dados mockados)
Variáveis de ambiente, CI/CD no Vercel e Railway
2
WhatsApp — Conexão e Inbox
Semanas 3–4
Evolution API SDK: cliente tipado, gerenciamento de instâncias
Tela de conexão: criar instância, gerar QR Code, status em tempo real
Webhook handler: receber, validar e salvar mensagens
Inbox de conversas: lista com busca, filtros e contagem de não lidos
Tela de conversa: histórico, envio manual, toggle IA on/off
3
IA Core — Atendimento + RAG
Semanas 5–6
Painel de configuração da IA: persona, empresa, FAQ, horários
BullMQ + Redis: fila de processamento de mensagens
Worker: carregar contexto + chamar OpenAI + enviar resposta
Knowledge Base: upload PDF/DOCX/TXT, chunking, embeddings via pgvector
RAG: busca semântica injetada no contexto do prompt
Tela de teste da IA: simular conversa antes de ativar
4
CRM + Qualificação de Leads
Semanas 7–8
Kanban CRM: 7 stages, drag-and-drop, movimentação automática via IA
Perfil do lead: histórico de conversa, score, tags, anotações
Fluxo de qualificação: perguntas configuráveis, lógica de score 0-100
Classificação automática: Frio / Morno / Quente
Integração Claude para análise de sentimento nas respostas
5
Follow-up + Billing + Admin
Semanas 9–10
Builder de sequências de follow-up: N passos, delay configurável por dia
Scheduler BullMQ: verificação por minuto, personalização de templates
Recuperação de leads: detectar abandonados, campanhas automáticas
Stripe: planos Starter/Pro/Business/Agency, portal do cliente
Limites por plano: controle de mensagens IA e instâncias WhatsApp
Painel admin: usuários, tenants, logs, consumo de IA
Analytics: gráficos de conversas, leads, receita recuperada
